async function fetchAccessToken() {
  const basic = Buffer.from(
    `${config.consumerKey}:${config.consumerSecret}`
  ).toString('base64');
  const { data } = await axios.get(
    `${config.baseUrl}/oauth/v1/generate?grant_type=client_credentials`,
    { headers: { Authorization: `Basic ${basic}` } }
  );
  cachedToken = data.access_token;
  tokenExpiresAt = Date.now() + (Number(data.expires_in) - 60) * 1000;
  return cachedToken;
}

export function darajaTimestamp(date = new Date()) {
  // Format: YYYYMMDDHHmmss — manual string padding
  const pad = (n) => String(n).padStart(2, '0');
  return date.getFullYear() + pad(date.getMonth() + 1) + ...;
}
export function stkPushPassword(shortcode, passkey, timestamp) {
  return Buffer.from(`${shortcode}${passkey}${timestamp}`).toString('base64');
}

  return data;
}

// utils/security.js — RSA certificate encryption (74 lines)
export function buildSecurityCredential() {
  const certPath = path.resolve(process.cwd(), config.certPath);
  if (!fs.existsSync(certPath)) {
    throw new Error(
      `Cert not found at ${certPath}. Download SandboxCertificate.cer...`
    );
  }
  const cert = fs.readFileSync(certPath);
  const encrypted = crypto.publicEncrypt(
    { key: cert, padding: crypto.constants.RSA_PKCS1_PADDING },
    Buffer.from(config.initiatorPassword, 'utf8')
  );
  return encrypted.toString('base64');
}

// daraja/b2c.js — B2C request (41 lines)
export async function b2cPayment({ amount, phone, commandID, ... }) {
  const securityCredential = buildSecurityCredential();
  const payload = {
    InitiatorName: config.initiatorName,
    SecurityCredential: securityCredential,
    CommandID: commandID,
    Amount: String(amount),
    PartyA: config.b2cShortcode,
    PartyB: normalizeMsisdn(phone),
    ResultURL: resultUrl,
    QueueTimeOutURL: queueTimeoutUrl,
    // ... more fields
  };
  const { data } = await http.post('/mpesa/b2c/v1/paymentrequest', payload);
  return data;
}

INTELLIGENCE IN EVERY CALL

Intelligence only M-Pesa can deliver

Every Daraja 4.0 call returns a payment receipt plus intelligence. Three capabilities ship embedded in every call — so a three-person fintech can launch a regulated product without a vendor stack, and a consumer app can personalise without ever seeing raw transaction data.

Capability 1

Identity & Verification

KYC in 60 seconds. One Daraja call returns identity confidence, device consistency, location check, and a risk score — grounded in GSM and financial data only M-Pesa has.

Use case: A three-person fintech in Nairobi approves a microloan in 60 seconds — one API call replaces their entire KYC vendor stack.

// One call. Borrower verified in 60 seconds. { confidence: 0.94, device_match: true, location_ok: true, risk: "low" }
Capability 2

Customer Intelligence

Consented signals, zero raw data. Apps consume derived patterns — spending rhythm, next predicted activity, primary categories — behind a scope-limited, revocable consent flow. No transaction history ever crosses the API boundary.

Use case: A consumer app warns you before cash runs short and flags subscriptions you've stopped using — without ever seeing your raw transactions.

Capability 3

Merchant Health Signal

A real-time categorical read on every merchant. Stable, growing, declining, at-risk — computed on two-sided visibility across 3M+ merchants that no global platform can match.

Use case: A wholesaler's AI agent checks merchant health before every supplier payment — financing priced in real time, no credit bureau needed.

// Real-time read on 3M+ merchants. { status: "growing", trend: "+12% MoM", risk_flag: null }
